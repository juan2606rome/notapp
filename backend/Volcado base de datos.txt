-- ─── CREAR TABLA ─────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS pokemon (
  id          INTEGER PRIMARY KEY,
  name        VARCHAR(100) NOT NULL UNIQUE,
  front_image TEXT,
  back_image  TEXT,
  shiny_image TEXT,
  height      INTEGER,
  weight      INTEGER,
  type        VARCHAR(50)
);
